19 def __init__(self, _username: str = "", _password: str = "", _agent: str = "TemporaryUserAgent", 

20 _download_directory: str = "./data/downloads/", _subtitle_language: str = ""): 

21 """ 

22 Constructor 

23 :param _username: username of the opensubtitles account 

24 :param _password: password of the opensubtitles account

26 If none is provided the default "TemporaryUserAgent" is used. This has query limits. 

27 :param _download_directory: download_directory to save the files 

28 """

32 self.__download_directory = _download_directory 

33 self.__user_agent = _agent

35 self.__subtitle_language = "eng" if _subtitle_language == "" else _subtitle_language

41 def check_server_status(self) -> CommandResponse: 

42 """ 

43 checks if the opensubtitles api endpoint is reachable. If the enpoint responds within 3 seconds the server_status is True, else False 

44 :return: connection CommandResponse 

45 """ 

46 _server_info = self.__proxy.ServerInfo() 

47 if _server_info['seconds'] <= 3.0: 

48 return CommandResponse(_successful=True, _message="is up and running") 

49 else:

53 """ 

54 logging in into opensubtitle with the given credentials. Sets the login token, which must be used in later communication. 

55 :return: login CommandResponse 

56 """ 

57 try: 

58 _login_result = self.__proxy.LogIn(self.__username, self.__password, self.__language, self.__user_agent) 

59 if _login_result['status'] == "200 OK": 

60 self.__login_token = _login_result['token'] 

61 return CommandResponse(_successful=True, _message="Log in successfully, and the token:" + self.__login_token) 

62 else:

140 def save_gz_format_in_directory(self, _subtitle, _directory_path): 

141 """ 

142 saves the file subtitleid.gz (encoded gz subtitle file) in the download directory.) 

143 """ 

144 with open(_directory_path, 'wb') as f: 

145 f.write(base64.b64decode(_subtitle))

147 def save_json_format_in_directory(self, _meta_subtitle, _directory_path): 

148 """ 

149 saves the file subtitleid.json (meta subtitle info) in the download directory.

151 with open(_directory_path, 'w', encoding='utf-8') as f:

171 def search_subtitles_by_id(self, _imdb_ids: [str]) -> [MetaSubtitle]: 

172 """ 

173 searches opensubtitle for a list of imdb ids. Returns information about found subtitles. 

174 :param _imdb_ids: list of imdb ids 

175 :return: list of MetaSubtitles 

176 """ 

177 _search_body = [] 

178 for _imdb_id in _imdb_ids: 

179 _search_body.append({'imdbid': _imdb_id, 'sublanguageid': self.__subtitle_language}) 

180 try: 

181 _subtitles = [] 

182 # limit for each movie the subtitle files 

183 for _imdb_movie in _search_body: 

184 _search_result = self.search_subtitle_from_proxy(_imdb_movie) 

185 if _search_result['status'] == "200 OK": 

186 for _r in _search_result['data']: 

187 _subtitles.append( 

188 MetaSubtitle(_movie_name=_r["MovieName"], _movie_release_name=_r["MovieReleaseName"], 

189 _info_format=_r["InfoFormat"], _sub_language_id=_r["SubLanguageID"] 

190 , _ISO639=_r["ISO639"], _movie_kind=_r["MovieKind"], 

191 _id_movie_imdb=_r["IDMovieImdb"], _id_subtitle_file=_r['IDSubtitleFile'], 

192 _sub_encoding=_r['SubEncoding'])) 

193 else: 

194 raise OpenSubtitleSearchError(_search_result['status'])
