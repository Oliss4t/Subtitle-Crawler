40 , _genres = _movie.get('genres'), _runtimes = _movie.get('runtimes'), _votes = _movie.get('votes'), _rating = _movie.get('rating') 

41 , _plot = _movie.get('plot outline'), _languages = _movie.get('languages'), _kind = _movie.get('kind'), _directors = _movie.get('directors')

58 def save_movie_infos_in_directory(self, _movie_infos: [MovieInfo]): 

59 """ 

60 saves the downloaded imdb information in the download directory 

61 and calls save_movie_image_in_directory to save the image in the download directory. 

62 The following formats will be saved: 

63 -moviename.json (meta subtitle info) 

64 """ 

65 try: 

66 for _info in _movie_infos: 

67 _working_directory_str = self.__download_directory + '/' + _info.id.strip('0') + '/' 

68 Path(_working_directory_str).mkdir(parents=True, exist_ok=True) 

69 _subtitle_path_json = Path(_working_directory_str, f"{_info.name}.json") 

70 _image_path_jpg = Path(_working_directory_str, f"{_info.name}.jpg") 

71 

72 self.save_movie_image_in_directory(_info.image_url, _image_path_jpg) 

73 

74 with open(_subtitle_path_json, 'w', encoding='utf-8') as f: 

75 serialized = jsonpickle.encode(_info) 

76 json.dump(json.loads(serialized), f, ensure_ascii=False, indent=4) 

77 

78 except ImdbSaveImageError as _image_e: 

79 raise _image_e 

80 except Exception as e: 

81 raise ImdbSaveInfoError(e.message if hasattr(e, 'massage') else None)
